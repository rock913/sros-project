#### Task 5.2: Statistics & Analytics APIs (Day 3-4)
**目标:** 实现 4 个核心统计 API 端点

**API 实现清单:**

1. **GET /stats/global** - 全局统计
   ```python
   # backend/src/agent/app.py
   
   class GlobalStatsResponse(BaseModel):
       total_sessions: int
       total_papers: int
       total_reports: int
       active_sessions: int  # status='active'
       avg_papers_per_session: float
       avg_reports_per_session: float
       most_common_sources: List[Dict[str, int]]  # {source: count}
       date_range: Dict[str, str]  # {first_paper, last_paper}
   
   @app.get("/stats/global", response_model=GlobalStatsResponse, tags=["Analytics"])
   async def get_global_stats():
       """Get overall platform statistics"""
       stats = db_manager.get_global_statistics()
       return GlobalStatsResponse(**stats)
   ```

2. **GET /stats/trends** - 时间趋势数据
   ```python
   class TrendDataPoint(BaseModel):
       date: str  # YYYY-MM-DD
       papers_count: int
       reports_count: int
       new_sessions: int
   
   class TrendsResponse(BaseModel):
       interval: str  # 'daily', 'weekly', 'monthly'
       data: List[TrendDataPoint]
   
   @app.get("/stats/trends", response_model=TrendsResponse, tags=["Analytics"])
   async def get_activity_trends(
       start_date: Optional[str] = Query(None),
       end_date: Optional[str] = Query(None),
       interval: str = Query('daily', regex='^(daily|weekly|monthly)$')
   ):
       """Get time-series activity data"""
       trends = db_manager.get_activity_trends(start_date, end_date, interval)
       return TrendsResponse(interval=interval, data=trends)
   ```

3. **GET /stats/keywords** - 关键词分析
   ```python
   class KeywordItem(BaseModel):
       keyword: str
       count: int
       sources: List[str]  # Which papers contain this keyword
   
   class KeywordsResponse(BaseModel):
       keywords: List[KeywordItem]
       total_unique_keywords: int
       extraction_method: str  # 'title_abstract_scan' or 'nlp_extraction'
   
   @app.get("/stats/keywords", response_model=KeywordsResponse, tags=["Analytics"])
   async def get_keyword_analysis(
       limit: int = Query(50, ge=1, le=200),
       source: Optional[str] = Query(None),
       session_id: Optional[str] = Query(None)
   ):
       """Extract and rank keywords from papers"""
       keywords = db_manager.get_keyword_frequencies(limit, source, session_id)
       return KeywordsResponse(
           keywords=keywords,
           total_unique_keywords=len(keywords),
           extraction_method='title_abstract_scan'
       )
   ```

4. **GET /stats/authors** - 作者网络数据
   ```python
   class AuthorNode(BaseModel):
       name: str
       paper_count: int
       co_authors: List[str]
   
   class AuthorNetworkResponse(BaseModel):
       nodes: List[AuthorNode]
       edges: List[Dict[str, Any]]  # [{source, target, weight}]
       total_authors: int
       collaboration_index: float  # avg co-authors per author
   
   @app.get("/stats/authors", response_model=AuthorNetworkResponse, tags=["Analytics"])
   async def get_author_network(
       min_papers: int = Query(1, ge=1),
       session_id: Optional[str] = Query(None)
   ):
       """Get author collaboration network data"""
       network = db_manager.get_author_network_data(min_papers, session_id)
       return AuthorNetworkResponse(**network)
   ```

   from apscheduler.schedulers.background import BackgroundScheduler
   from agent.db_manager import cache_global_stats, cache_keyword_stats
   
   scheduler = BackgroundScheduler()
   
   # Every hour at :00
   scheduler.add_job(cache_global_stats, 'cron', hour='*', minute=0)

```typescript
// vscode-extension/src/analytics/AnalyticsDashboard.tsx

import React, { useEffect, useState } from 'react';
import { GlobalStatsCard } from './components/GlobalStatsCard';
import { ActivityTrendChart } from './components/ActivityTrendChart';
import { SourceDistributionChart } from './components/SourceDistributionChart';
import { getGlobalStats, getActivityTrends, getKeywords } from '../api';

export const AnalyticsDashboard: React.FC = () => {
    const [stats, setStats] = useState<GlobalStats | null>(null);
    const [trends, setTrends] = useState<TrendDataPoint[]>([]);
    const [interval, setInterval] = useState<'daily' | 'weekly' | 'monthly'>('daily');
    const [loading, setLoading] = useState(true);
    
    useEffect(() => {
        loadAllData();
    }, [interval]);
    
    const loadAllData = async () => {
        setLoading(true);
        try {
            const [statsData, trendsData] = await Promise.all([
                getGlobalStats(),
                getActivityTrends(interval)
            ]);
            setStats(statsData);
            setTrends(trendsData);
        } catch (error) {
            console.error('Failed to load analytics:', error);
        } finally {
            setLoading(false);
        }
    };
    
    if (loading) {
        return <div className="loading">Loading analytics...</div>;
    }
    
    return (
        <div className="analytics-dashboard">
            <header>
                <h1>Research Analytics Dashboard</h1>
                <button onClick={loadAllData}>Refresh</button>
            </header>
            
            {stats && <GlobalStatsCard stats={stats} />}
            
            <div className="charts-section">
                <div className="chart-container">
                    <div className="interval-selector">
                        <button onClick={() => setInterval('daily')}>Daily</button>
                        <button onClick={() => setInterval('weekly')}>Weekly</button>
                        <button onClick={() => setInterval('monthly')}>Monthly</button>
                    </div>
                    <ActivityTrendChart data={trends} interval={interval} />
                </div>
                
                {stats && (
                    <div className="chart-container">
                        <SourceDistributionChart sources={stats.most_common_sources} />
                    </div>
                )}
            </div>
        </div>
    );
};
```
