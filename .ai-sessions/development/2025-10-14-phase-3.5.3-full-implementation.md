#### Step 1.2.3: 实现Webview消息处理

**修改文件**: `vscode-extension/src/extension.ts`

**在 `showControlPanel` command中添加消息处理**:
```typescript
// Handle messages from webview
panel.webview.onDidReceiveMessage(async (message) => {
  switch (message.command) {
    case 'startResearch':
      const topic = message.topic;
      const threadId = message.threadId; // optional
      
      try {
        await startResearchStream(topic, {
          onStarted: (data) => {
            panel.webview.postMessage({
              type: 'progress',
              message: `🚀 Research started (Session: ${data.session_id})`
            });
          },
          onProgress: (data) => {
            // Extract node name
            const nodeNames = Object.keys(data.chunk);
            const nodeName = nodeNames.length > 0 ? nodeNames[0] : 'unknown';
            
            panel.webview.postMessage({
              type: 'progress',
              message: `⚙️  Processing: ${nodeName}`,
              data: data.chunk
            });
          },
          onComplete: (data) => {
            panel.webview.postMessage({
              type: 'complete',
              message: '✅ Research completed!'
            });
            
            // Refresh TreeViews
            assetLibraryProvider.refresh();
            manuscriptProvider.refresh();
            
            vscode.window.showInformationMessage('Research completed successfully!');
          },
          onError: (error) => {
            panel.webview.postMessage({
              type: 'error',
              message: `❌ Error: ${error}`
            });
            vscode.window.showErrorMessage(`Research failed: ${error}`);
          }
        }, threadId);
        
      } catch (error) {
        vscode.window.showErrorMessage(`Failed to start research: ${error}`);
      }
      break;
  }
});
```
